68 static void
69 update_result (char *got, const char *expected, const char *test)
70 {
71  int failed, n;
72 
73  failed = strcmp(got, expected);
74  if (failed) {
75  for (n = 0; n < strlen(expected); n++) {
76  if (got[n] != expected[n]) {
77  printf("Index %d: expected=%c got=%c\n", n, expected[n],
78  got[n]);
79  }
80  }
81  }
82  if (!failed) {
83  printf("\nPASS: %s\n", test);
84  num_passed++;
85  } else {
86  printf("\nFAIL: %s\n", test);
87  num_failed++;
88  }
89  fflush(stdout);
90 }
