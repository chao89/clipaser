55 static void
56 cparser_term_set_canonical (cparser_t *parser, int enable)
57 {
58  static struct termios old_term;
59  static int old_term_set = 0;
60  struct termios new_term;
61 
62  if (!VALID_PARSER(parser)) return;
63 
64  if (enable) {
65  assert(old_term_set);
66  tcsetattr(STDIN_FILENO, TCSANOW, &old_term);
67  old_term_set = 0;
68  } else {
69  assert(0 == STDIN_FILENO);
70  tcgetattr(STDIN_FILENO, &old_term);
71  new_term = old_term;
72  new_term.c_lflag &= ~(ICANON | ECHO);
73  tcsetattr(STDIN_FILENO, TCSANOW, &new_term);
74  old_term_set = 1;
75  }
76 }

78 static void
79 cparser_unix_getch (cparser_t *parser, int *ch, cparser_char_t *type)
80 {
81  assert(VALID_PARSER(parser) && ch && type);
82  *type = CPARSER_CHAR_UNKNOWN;
83  *ch = getchar();
84  if ('' == *ch) {
85  *ch = getchar();
86  if ('[' == *ch) {
87  *ch = getchar();
88  switch (*ch) {
89  case 'A':
90  *type = CPARSER_CHAR_UP_ARROW;
91  break;
92  case 'B':
93  *type = CPARSER_CHAR_DOWN_ARROW;
94  break;
95  case 'C':
96  *type = CPARSER_CHAR_RIGHT_ARROW;
97  break;
98  case 'D':
99  *type = CPARSER_CHAR_LEFT_ARROW;
100  break;
101  }
102  }
103 #ifdef CPARSER_EMACS_BINDING
104  } else if (CTRL_P == (*ch)) {
105  *type = CPARSER_CHAR_UP_ARROW;
106  } else if (CTRL_N == (*ch)) {
107  *type = CPARSER_CHAR_DOWN_ARROW;
108  } else if (CTRL_A == (*ch)) {
109  *type = CPARSER_CHAR_FIRST;
110  } else if (CTRL_E == (*ch)) {
111  *type = CPARSER_CHAR_LAST;
112 #endif /* EMACS_BINDING */
113  } else if (isalnum(*ch) || ('\n' == *ch) ||
114  ispunct(*ch) || (' ' == *ch) ||
115  (*ch == parser->cfg.ch_erase) ||
116  (*ch == parser->cfg.ch_del) ||
117  (*ch == parser->cfg.ch_help) ||
118  (*ch == parser->cfg.ch_complete)) {
119  *type = CPARSER_CHAR_REGULAR;
120  }
121 }
