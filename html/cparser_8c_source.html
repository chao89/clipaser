66 static void
67 cparser_help_print_node (cparser_t *parser, cparser_node_t *node,
68  const int add_lf, const int print_desc)
69 {
70  assert(parser && node);
71  if (!NODE_USABLE(parser, node)) {
72  return;
73  }
74  if (add_lf) {
75  parser->cfg.printc(parser, '\n');
76  }
77  switch (node->type) {
78  case CPARSER_NODE_ROOT:
79  assert(0); /* this should never happen */
80  case CPARSER_NODE_END:
81  parser->cfg.prints(parser, "<LF>");
82  break;
83  case CPARSER_NODE_LIST:
84  parser->cfg.prints(parser, "[ ");
85  cparser_list_node_t *lnode = (cparser_list_node_t *)node->param;
86  assert(lnode);
87  while (lnode) {
88  parser->cfg.prints(parser, lnode->keyword);
89  lnode = lnode->next;
90  if (lnode) {
91  parser->cfg.prints(parser, " | ");
92  }
93  }
94  parser->cfg.prints(parser, " ]");
95  break;
96  default:
97  parser->cfg.prints(parser, node->param);
98  if (print_desc && node->desc) {
99  parser->cfg.prints(parser, " - ");
100  parser->cfg.prints(parser, node->desc);
101  }
102  break;
103  }
104 }

180 static cparser_result_t
181 cparser_execute_cmd (cparser_t *parser)
182 {
183  int do_echo;
184  cparser_result_t rc = CPARSER_OK;
185  assert(VALID_PARSER(parser));
186 
187  /*
188  * Enter a command. There are three possibilites:
189  * 1. If we are in WHITESPACE state, we check if there is
190  * only one child of keyword type. If yes, we recurse
191  * into it and repeat until either: a) there is more than
192  * one choice, b) We are at a END node. If there is more
193  * than one choice, we look for an END node. In either
194  * case, if an END node is found, we execute the action
195  * function.
196  *
197  * 2. If we are in TOKEN state, we check if we have an unique
198  * match. If yes, re recurse into it and repeat just
199  * like WHITESPACE state until we find an END node.
200  *
201  * 3. If we are in ERROR state, we print out an error.
202  *
203  * Afterward, we reset the parser state and move to the
204  * next line buffer.
205  */
206  if ((CPARSER_STATE_TOKEN == parser->state) ||
207  (CPARSER_STATE_WHITESPACE == parser->state)) {
208  cparser_node_t *child;
209 
210  if (CPARSER_STATE_TOKEN == parser->state) {
211  cparser_token_t *token;
212  cparser_node_t *match;
213  int is_complete;
214 
215  token = CUR_TOKEN(parser);
216  if ((1 <= cparser_match(parser, token->buf, token->token_len,
217  parser->cur_node, &match,
218  &is_complete)) &&
219  (is_complete)) {
220  cparser_complete_fn fn = cparser_complete_fn_tbl[match->type];
221  if (fn) {
222  fn(parser, match, token->buf, token->token_len);
223  }
224  rc = cparser_input(parser, ' ', CPARSER_CHAR_REGULAR);
225  assert(CPARSER_OK == rc);
226  } else {
227  cparser_print_error(parser, "Incomplete command\n");
228  rc = CPARSER_ERR_INCOMP_CMD;
229 
230  /* Reset the internal buffer, state and cur_node */
231  cparser_record_command(parser, rc);
232  cparser_fsm_reset(parser);
233  cparser_print_prompt(parser);
234  return rc;
235  }
236  }
237 
238  /* Look for a single keyword node child */
239  child = parser->cur_node->children;
240  assert(child);
241  while ((CPARSER_NODE_KEYWORD == child->type) &&
242  NODE_USABLE(parser, child) && (!child->sibling)) {
243  cparser_token_t *token = CUR_TOKEN(parser);
244  cparser_complete_keyword(parser, child, token->buf, token->token_len);
245  rc = cparser_input(parser, ' ', CPARSER_CHAR_REGULAR);
246  assert(CPARSER_OK == rc);
247 
248  child = parser->cur_node->children;
249  assert(child);
250  }
251 
252  /* Look for an end node */
253  child = parser->cur_node->children;
254  while ((NULL != child) &&
255  !((CPARSER_NODE_END == child->type) && NODE_USABLE(parser, child))) {
256  child = child->sibling;
257  }
258  if (child) {
259  assert(CPARSER_NODE_END == child->type);
260 
261  /* Execute the glue function */
262  parser->cur_node = child;
263  parser->cfg.printc(parser, '\n');
264  rc = ((cparser_glue_fn)child->param)(parser);
265  } else {
266  if (parser->token_tos) {
267  cparser_print_error(parser, "Incomplete command\n");
268  rc = CPARSER_ERR_INCOMP_CMD;
269  }
270 
271  /* Reset FSM states and advance to the next line */
272  cparser_record_command(parser, rc);
273  cparser_fsm_reset(parser);
274  if (CPARSER_OK == rc) {
275  /* This is just a blank line */
276  parser->cfg.printc(parser, '\n');
277  }
278  cparser_print_prompt(parser);
279  return rc;
280  }
281  } else if (CPARSER_STATE_ERROR == parser->state) {
282  cparser_print_error(parser, "Parse error\n");
283  rc = CPARSER_ERR_PARSE_ERR;
284  }
285 
286  /* Reset FSM states and advance to the next line */
287  cparser_record_command(parser, rc);
288  cparser_fsm_reset(parser);
289  if (!cparser_is_user_input(parser, &do_echo)) {
290  cparser_print_prompt(parser);
291  }
292  return rc;
293 }

295 static cparser_result_t
296 cparser_match_prefix (const cparser_t *parser, const char *token,
297  const int token_len, const cparser_node_t *parent,
298  const char ch, const int offset)
299 {
300  int local_is_complete;
301  cparser_node_t *child;
302  cparser_result_t rc;
303 
304  assert(parent && ch);
305  for (child = parent->children; NULL != child; child = child->sibling) {
306  if (!NODE_USABLE(parser, child)) {
307  continue;
308  }
309  rc = cparser_match_fn_tbl[child->type](token, token_len, child,
310  &local_is_complete);
311  if (CPARSER_NOT_OK == rc) {
312  continue;
313  }
314  if (CPARSER_NODE_KEYWORD != child->type) {
315  return CPARSER_NOT_OK;
316  }
317 
318  /* There is a match. Make sure that it is part of this node as well */
319  if (*((char *)child->param + offset) != ch) {
320  return CPARSER_NOT_OK;
321  }
322  }
323  return CPARSER_OK;
324 }

331 static cparser_result_t
332 cparser_help (cparser_t *parser)
333 {
334  cparser_node_t *node;
335  cparser_token_t *token;
336  int local_is_complete;
337 
338  assert(VALID_PARSER(parser));
339  if (CPARSER_STATE_WHITESPACE == parser->state) {
340  /* Just print out every children */
341  for (node = parser->cur_node->children; NULL != node;
342  node = node->sibling) {
343  cparser_help_print_node(parser, node, 1, 1);
344  }
345  } else if (CPARSER_STATE_ERROR == parser->state) {
346  /*
347  * We have some problem parsing. Just print out the last known
348  * good parse point and list the valid options.
349  */
350  cparser_print_error(parser, "Last known good parse point.");
351  for (node = parser->cur_node->children; NULL != node;
352  node = node->sibling) {
353  cparser_help_print_node(parser, node, 1, 1);
354  }
355  } else {
356  /* We have a partial match */
357  node = parser->cur_node->children;
358  token = CUR_TOKEN(parser);
359  for (node = parser->cur_node->children; NULL != node;
360  node = node->sibling) {
361  if (!NODE_USABLE(parser, node)) {
362  continue;
363  }
364  if (CPARSER_OK ==
365  cparser_match_fn_tbl[node->type](token->buf, token->token_len,
366  node, &local_is_complete)) {
367  cparser_help_print_node(parser, node, 1, 1);
368  }
369  }
370  }
371  cparser_line_print(parser, 1, 1);
372  return CPARSER_OK;
373 }

383 static int
384 cparser_complete_one_level (cparser_t *parser)
385 {
386  cparser_token_t *token;
387  cparser_node_t *match;
388  int is_complete = 0, num_matches, keep_going = 0, rc;
389  char *ch_ptr;
390 
391  switch (parser->state) {
392  case CPARSER_STATE_ERROR:
393  /* If we are in ERROR, there cannot be a match. So, just quit */
394  parser->cfg.printc(parser, '\a');
395  break;
396  case CPARSER_STATE_WHITESPACE:
397  if (parser->cur_node && parser->cur_node->children &&
398  !parser->cur_node->children->sibling &&
399  (CPARSER_NODE_KEYWORD == parser->cur_node->children->type)) {
400  ch_ptr = parser->cur_node->children->param;
401  while (*ch_ptr) {
402  rc = cparser_input(parser, *ch_ptr, CPARSER_CHAR_REGULAR);
403  assert(CPARSER_OK == rc);
404  ch_ptr++;
405  }
406  rc = cparser_input(parser, ' ', CPARSER_CHAR_REGULAR);
407  assert(CPARSER_OK == rc);
408  } else {
409  /*
410  * If we are in WHITESPACE, just dump all children. Since there is no
411  * way any token can match to a NULL string.
412  */
413  cparser_help(parser);
414  }
415  break;
416  case CPARSER_STATE_TOKEN:
417  {
418  /* Complete a command */
419  token = CUR_TOKEN(parser);
420  num_matches = cparser_match(parser, token->buf, token->token_len,
421  parser->cur_node, &match,
422  &is_complete);
423  if ((1 == num_matches) && (is_complete)) {
424  cparser_complete_fn fn = cparser_complete_fn_tbl[match->type];
425  /*
426  * If the only matched node is a keyword, we feel the rest of
427  * keyword in. Otherwise, we assume this parameter is complete
428  * and just insert a space.
429  */
430  if (fn) {
431  fn(parser, match, token->buf, token->token_len);
432  }
433  rc = cparser_input(parser, ' ', CPARSER_CHAR_REGULAR);
434  assert(CPARSER_OK == rc);
435 
436  keep_going = 1;
437  } else {
438  int offset, orig_offset;
439  /*
440  * If we have more than one match, we should try to complete
441  * as much as possible. To do that, we grab the node in the
442  * (first) matched node and check that the next character
443  * from it is common among all matched nodes. If it is common
444  * to all matched nodes, we continue to feed them into the
445  * parser. However, this is only useful for keywords. If there
446  * is a parameter token in the match, we automatically abort.
447  */
448  offset = orig_offset = token->token_len;
449  ch_ptr = match->param + token->token_len;
450  while (('\0' != *ch_ptr) &&
451  (CPARSER_OK ==
452  cparser_match_prefix(parser, token->buf, token->token_len,
453  parser->cur_node, *ch_ptr,
454  offset))) {
455  rc = cparser_input(parser, *ch_ptr, CPARSER_CHAR_REGULAR);
456  assert(CPARSER_OK == rc);
457  ch_ptr++;
458  offset++;
459  }
460  if (orig_offset == offset) {
461  /* If there is no common prefix at all, just display help */
462  cparser_help(parser);
463  }
464  }
465  break;
466  }
467  default: assert(0);
468  }
469 
470  return keep_going;
471 }

473 cparser_result_t
474 cparser_input (cparser_t *parser, char ch, cparser_char_t ch_type)
475 {
476  int n, do_echo;
477  cparser_result_t rc;
478 
479  if (!VALID_PARSER(parser)) {
480  return CPARSER_ERR_INVALID_PARAMS;
481  }
482 
483  if (cparser_is_user_input(parser, &do_echo)) {
484  /* Process user input */
485  if (CPARSER_CHAR_REGULAR != ch_type) {
486  return CPARSER_OK;
487  }
488  if ('\n' == ch) {
489  /* We have a complete input. Call the callback. */
490  assert(parser->user_input_cb);
491  parser->user_buf[parser->user_buf_count] = '\0';
492  rc = parser->user_input_cb(parser, parser->user_buf,
493  parser->user_buf_count);
494  cparser_input_reset(parser);
495  cparser_print_prompt(parser);
496  return rc;
497  }
498 
499  if ((parser->cfg.ch_erase == ch) || (parser->cfg.ch_del == ch)) {
500  if (parser->user_buf_count > 0) {
501  parser->user_buf_count--;
502  }
503  if (parser->user_do_echo) {
504  parser->cfg.printc(parser, '\b');
505  }
506  } else if ((parser->user_buf_count + 1) < parser->user_buf_size) {
507  parser->user_buf[parser->user_buf_count] = ch;
508  parser->user_buf_count++;
509  if (parser->user_do_echo) {
510  parser->cfg.printc(parser, ch);
511  }
512  }
513  return CPARSER_OK;
514  }
515 
516  switch (ch_type) {
517  case CPARSER_CHAR_REGULAR:
518  {
519  if ((parser->cfg.ch_complete == ch) ||
520  (parser->cfg.ch_help == ch)) {
521  /*
522  * Completion and help character do not go into the line
523  * buffer. So, do nothing.
524  */
525  break;
526  }
527  if ((parser->cfg.ch_erase == ch) || (parser->cfg.ch_del == ch)) {
528  rc = cparser_line_delete(parser);
529  assert(CPARSER_ERR_INVALID_PARAMS != rc);
530  if (CPARSER_ERR_NOT_EXIST == rc) {
531  return CPARSER_OK;
532  }
533  } else if ('\n' == ch) {
534  /* Put the rest of the line into parser FSM */
535  for (n = cparser_line_current(parser);
536  n < cparser_line_last(parser); n++) {
537  rc = cparser_fsm_input(parser, cparser_line_char(parser, n));
538  assert(CPARSER_OK == rc);
539  }
540  } else {
541  (void)cparser_line_insert(parser, ch);
542  }
543  break;
544  }
545  case CPARSER_CHAR_UP_ARROW:
546  {
547  rc = cparser_line_prev_line(parser);
548  assert(CPARSER_OK == rc);
549 
550  /* Reset the token stack and re-enter the command */
551  cparser_fsm_reset(parser);
552  for (n = 0; n < cparser_line_current(parser); n++) {
553  rc = cparser_fsm_input(parser, cparser_line_char(parser, n));
554  assert(CPARSER_OK == rc);
555  }
556 
557  return CPARSER_OK;
558  }
559  case CPARSER_CHAR_DOWN_ARROW:
560  {
561  rc = cparser_line_next_line(parser);
562  assert(CPARSER_OK == rc);
563 
564  /* Reset the token stack and re-enter the command */
565  cparser_fsm_reset(parser);
566  for (n = 0; n < cparser_line_current(parser); n++) {
567  rc = cparser_fsm_input(parser, cparser_line_char(parser, n));
568  assert(CPARSER_OK == rc);
569  }
570 
571  return CPARSER_OK;
572  }
573  case CPARSER_CHAR_LEFT_ARROW:
574  {
575  ch = cparser_line_prev_char(parser);
576  if (!ch) {
577  parser->cfg.printc(parser, '\a');
578  return CPARSER_OK;
579  }
580  break;
581  }
582  case CPARSER_CHAR_RIGHT_ARROW:
583  {
584  ch = cparser_line_next_char(parser);
585  if (!ch) {
586  parser->cfg.printc(parser, '\a');
587  return CPARSER_OK;
588  }
589  break;
590  }
591  case CPARSER_CHAR_FIRST:
592  {
593  do {
594  ch = cparser_line_prev_char(parser);
595  if (ch) {
596  cparser_fsm_input(parser, ch);
597  }
598  } while (ch);
599  return CPARSER_OK;
600  }
601  case CPARSER_CHAR_LAST:
602  {
603  do {
604  ch = cparser_line_next_char(parser);
605  if (ch) {
606  cparser_fsm_input(parser, ch);
607  }
608  } while (ch);
609  return CPARSER_OK;
610  }
611  default:
612  {
613  /* An unknown character. Alert and continue */
614  parser->cfg.printc(parser, '\a');
615  return CPARSER_NOT_OK;
616  }
617  } /* switch (ch_type) */
618 
619  /* Handle special characters */
620  if (ch == parser->cfg.ch_complete) {
621  while (cparser_complete_one_level(parser));
622  return CPARSER_OK;
623  } else if (ch == parser->cfg.ch_help) {
624  /* Ask for context sensitve help */
625  cparser_help(parser);
626  return CPARSER_OK;
627  } else if ('\n' == ch) {
628  rc = cparser_execute_cmd(parser);
629  cparser_line_advance(parser);
630  return rc;
631  }
632 
633  return cparser_fsm_input(parser, (char)ch);
634 }

636 cparser_result_t
637 cparser_run (cparser_t *parser)
638 {
639  int ch;
640  cparser_char_t ch_type = 0;
641 
642  if (!VALID_PARSER(parser)) return CPARSER_ERR_INVALID_PARAMS;
643 
644  parser->cfg.io_init(parser);
645  cparser_print_prompt(parser);
646  parser->done = 0;
647 
648  while (!parser->done) {
649  parser->cfg.getch(parser, &ch, &ch_type);
650  cparser_input(parser, ch, ch_type);
651  } /* while not done */
652 
653  parser->cfg.io_cleanup(parser);
654 
655  return CPARSER_OK;
656 }

658 cparser_result_t
659 cparser_init (cparser_cfg_t *cfg, cparser_t *parser)
660 {
661  int n;
662 
663  if (!parser || !cfg || !cfg->root || !cfg->ch_erase) {
664  return CPARSER_ERR_INVALID_PARAMS;
665  }
666 
667  parser->cfg = *cfg;
668  parser->cfg.prompt[CPARSER_MAX_PROMPT-1] = '\0';
669 
670  /* Initialize sub-mode states */
671  parser->root_level = 0;
672  parser->root[0] = parser->cfg.root;
673  snprintf(parser->prompt[0], sizeof(parser->prompt[0]), "%s",
674  parser->cfg.prompt);
675  for (n = 0; n < CPARSER_MAX_NESTED_LEVELS; n++) {
676  parser->context.cookie[n] = NULL;
677  }
678  parser->context.parser = parser;
679 
680  /* Initialize line buffering states */
681  parser->max_line = 0;
682  parser->cur_line = 0;
683  for (n = 0; n < CPARSER_MAX_LINES; n++) {
684  cparser_line_reset(&parser->lines[n]);
685  }
686 
687  /* Initialize parser FSM state */
688  cparser_fsm_reset(parser);
689  parser->is_privileged_mode = 0;
690 
691  /* Clear the user input state */
692  cparser_input_reset(parser);
693 
694  return CPARSER_OK;
695 }

707 cparser_result_t
708 cparser_submode_enter (cparser_t *parser, void *cookie, char *prompt)
709 {
710  cparser_node_t *new_root;
711 
712  if (!parser) {
713  return CPARSER_ERR_INVALID_PARAMS;
714  }
715  if ((CPARSER_MAX_NESTED_LEVELS-1) == parser->root_level) {
716  return CPARSER_NOT_OK;
717  }
718  parser->root_level++;
719  new_root = parser->cur_node->children;
720  assert(new_root);
721  assert(CPARSER_NODE_ROOT == new_root->type);

743 cparser_result_t
744 cparser_load_cmd (cparser_t *parser, char *filename)
745 {
746  FILE *fp;
747  char buf[128];
748  size_t rsize, n;
749  int fd, indent = 0, last_indent = -1, new_line = 1, m, line_num = 0;
750 
751  if (!VALID_PARSER(parser) || !filename) {
752  return CPARSER_ERR_INVALID_PARAMS;
753  }
754 
755  fd = parser->cfg.fd;
756  parser->cfg.fd = -1;
757 
758  fp = fopen(filename, "r");
759  if (!fp) {
760  return CPARSER_NOT_OK;
761  }
762 
763  cparser_fsm_reset(parser);
764  while (!feof(fp)) {
765  rsize = fread(buf, 1, sizeof(buf), fp);
766  for (n = 0; n < rsize; n++) {
767  /* Examine the input characters to maintain indent level */
768  if ('\n' == buf[n]) {
769  cparser_result_t rc;
770  char buf[128];
771 
772  line_num++;
773  indent = 0;
774  new_line = 1;
775  rc = cparser_execute_cmd(parser);
776  if (CPARSER_OK == rc) {
777  continue;
778  }
779  parser->cfg.fd = fd;
780  switch (rc) {
781  case CPARSER_ERR_PARSE_ERR:
782  snprintf(buf, sizeof(buf), "Line %d: Parse error.\n", line_num);
783  parser->cfg.prints(parser, buf);
784  break;
785  case CPARSER_ERR_INCOMP_CMD:
786  snprintf(buf, sizeof(buf), "Line %d: Incomplete command.\n", line_num);
787  parser->cfg.prints(parser, buf);
788  break;
789  default:
790  assert(0);
791  }
792  return CPARSER_NOT_OK;
793  } else if (' ' == buf[n]) {
794  if (new_line) {
795  indent++;
796  }
797  } else {
798  if (new_line) {
799  new_line = 0;
800  if (indent < last_indent) {
801  for (m = indent; m < last_indent; m++) {
802  if (CPARSER_OK != cparser_submode_exit(parser)) {
803  break;
804  }
805  cparser_fsm_reset(parser);
806  }
807  }
808  last_indent = indent;
809  }
810  }
811  (void)cparser_fsm_input(parser, buf[n]);
812  }
813  }
814  fclose(fp);
815 
816  while (parser->root_level) {
817  (void)cparser_submode_exit(parser);
818  cparser_fsm_reset(parser);
819  }
820  parser->cfg.fd = fd;
821  return CPARSER_OK;
822 }

826  cparser_walker_fn pre_fn, cparser_walker_fn post_fn,

845  }

848  rc = post_fn(parser, node, cookie);
849  if (CPARSER_OK != rc) {

851  }
852  }

909 static cparser_result_t
910 cparser_help_post_walker (cparser_t *parser, cparser_node_t *node, void *cookie)
911 {
912  help_stack_t *hs = (help_stack_t *)cookie;
913  int n, do_print;
914 
915  assert(parser && node && hs);
916  if ((CPARSER_NODE_END == node->type) &&
917  (!(node->flags & CPARSER_NODE_FLAGS_OPT_PARTIAL))) {
918  do_print = 0;
919  if (hs->filter) {
920  /* We have a filter string. Check if it matches any keyword */
921  for (n = 0; n < hs->tos; n++) {
922  if (CPARSER_NODE_LIST == hs->nodes[n]->type) {
923  /* LIST node requires an extra walk of all keywords in the list */
924  cparser_list_node_t *lnode =
925  (cparser_list_node_t *)hs->nodes[n]->param;
926  assert(lnode);
927  while (lnode) {
928  if (strstr(lnode->keyword, hs->filter)) {
929  do_print = 1;
930  break;
931  }
932  lnode = lnode->next;
933  }
934  if (do_print) {
935  break;
936  }
937  }
938  if (CPARSER_NODE_KEYWORD != hs->nodes[n]->type) {
939  continue;
940  }
941  if (strstr(hs->nodes[n]->param, hs->filter)) {
942  do_print = 1; /* Yes, print it */
943  break;
944  }
945  }
946  if (!NODE_USABLE(parser, node)) {
947  do_print = 0; /* may match the filter but not usable */
948  }
949  } else if (NODE_USABLE(parser, node)) {
950  do_print = 1;
951  }
952  if (do_print) {
953  cparser_node_t *cur_node;
954  int m, num_braces = 0;
955 
956  if (node->desc) {
957  parser->cfg.prints(parser, node->desc);
958  parser->cfg.prints(parser, "\r\n ");
959  } else {
960  parser->cfg.prints(parser, "\r ");
961  }
962  for (n = 0; n < hs->tos; n++) {
963  cur_node = hs->nodes[n];
964  if ((CPARSER_NODE_ROOT == cur_node->type) ||
965  (CPARSER_NODE_END == cur_node->type)) {
966  continue;
967  }
968  if (cur_node->flags & CPARSER_NODE_FLAGS_OPT_START) {
969  parser->cfg.prints(parser, "{ ");
970  num_braces++;
971  }
972  cparser_help_print_node(parser, cur_node, 0, 0);
973  parser->cfg.printc(parser, ' ');
974  if (cur_node->flags & CPARSER_NODE_FLAGS_OPT_END) {
975  for (m = 0; m < num_braces; m++) {
976  parser->cfg.prints(parser, "} ");
977  }
978  }
979  }
980  parser->cfg.prints(parser, "\r\n\n");
981  }
982  }
983 
984  /* Pop the stack */
985  hs->tos--;
986  return CPARSER_OK;
987 }
