56 cparser_result_t
57 cparser_line_insert (cparser_t *parser, char ch)
58 {
59  int n;
60  cparser_line_t *line;
61 
62  if (!VALID_PARSER(parser) || !ch) {
63  return CPARSER_ERR_INVALID_PARAMS;
64  }
65  line = &parser->lines[parser->cur_line];
66  if (CPARSER_MAX_LINE_SIZE <= line->last) {
67  return CPARSER_ERR_OUT_OF_RES;
68  }
69 
70  /* Move all characters from current to last back by 1 */
71  for (n = line->last; n > line->current; n--) {
72  line->buf[n] = line->buf[n-1];
73  }
74  line->buf[++line->last] = '\0';
75 
76  /*
77  * Insert the new character and update the line display. We do not
78  * have full curse support here. Instead, we simply assume all
79  * characters are on the same line and use backspace to move the
80  * cursor.
81  */
82  line->buf[line->current] = ch;
83  parser->cfg.printc(parser, ch);
84  line->current++; /* update current position */
85  parser->cfg.prints(parser, LINE_CURRENT(line));
86 
87  /* Move cursor back to the current position */
88  for (n = line->current; n < line->last; n++) {
89  parser->cfg.printc(parser, '\b');
90  }
91 
92  return CPARSER_OK;
93 }

95 cparser_result_t
96 cparser_line_delete (cparser_t *parser)
97 {
98  cparser_line_t *line;
99  int n;
100 
101  if (!VALID_PARSER(parser)) {
102  return CPARSER_ERR_INVALID_PARAMS;
103  }
104  line = &parser->lines[parser->cur_line];
105  assert(line->current <= line->last);
106  if (!line->last || !line->current) {
107  /* Line is empty or we're at the beginning of the line */
108  return CPARSER_ERR_NOT_EXIST;
109  }
110 
111  /* Move all character after current position back by one */
112  for (n = line->current; n < line->last; n++) {
113  line->buf[n-1] = line->buf[n];
114  }
115  line->current--;
116  line->last--;
117  line->buf[line->last] = '\0';
118 
119  /* Update the display */
120  parser->cfg.printc(parser, '\b');
121  parser->cfg.prints(parser, LINE_CURRENT(line));
122  parser->cfg.prints(parser, " \b");
123  for (n = line->current; n < line->last; n++) {
124  parser->cfg.printc(parser, '\b');
125  }
126  return CPARSER_OK;
127 }

189 
190  assert(VALID_PARSER(parser));
191  line = &parser->lines[parser->cur_line];
192  if (line->last == line->current) {

195  }
196 
197  retval = line->buf[line->current];

222 cparser_result_t
223 cparser_line_next_line (cparser_t *parser)
224 {
225  int n;
226 
227  if (!VALID_PARSER(parser)) {
228  return CPARSER_ERR_INVALID_PARAMS;
229  }
230 
231  /* Erase the current line */
232  for (n = 0; n < cparser_line_last(parser); n++) {
233  parser->cfg.prints(parser, "\b \b");
234  }
235 
236  /* Go to the next line */
237  parser->cur_line++;
238  if (parser->max_line < parser->cur_line) {
239  parser->cur_line = 0;
240  }
241 
242  /* Print out the new line */
243  cparser_line_print(parser, 0, 0);
244 
245  return CPARSER_OK;
246 }
