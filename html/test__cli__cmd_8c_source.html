89 static cparser_result_t
90 test_employee_add (employee_t *roster, const uint32_t id)
91 {
92  int n;
93 
94  assert(roster && id);
95 
96  /* Look for an unused slot */
97  for (n = 0; n < MAX_EMPLOYEES; n++) {
98  if (!roster[n].id) {
99  /* Found one. Use it */
100  memset(&roster[n], 0, sizeof(roster[n]));
101  roster[n].id = id;
102  return CPARSER_OK;
103  }
104  }
105 
106  return CPARSER_ERR_OUT_OF_RES;
107 }

455 cparser_result_t
456 cparser_cmd_employee_id (cparser_context_t *context, uint32_t *id)
457 {
458  employee_t *new_emp;
459  char prompt[CPARSER_MAX_PROMPT];
460 
461  assert(context && id && (*id));
462 
463  /* Check that there is no existing employee by that name */
464  new_emp = test_employee_search(roster, *id);
465  if (!new_emp) {
466  /* Add a new record with that name */
467  if (CPARSER_OK != test_employee_add(roster, *id)) {
468  PRINTF("Fail to add employee.\n");
469  return CPARSER_NOT_OK;
470  }
471  new_emp = test_employee_search(roster, *id);
472  }
473 
474  /* Enter the submode */
475  assert(new_emp);
476  snprintf(prompt, CPARSER_MAX_PROMPT, "0x%08X: ", *id);
477  return cparser_submode_enter(context->parser, new_emp, prompt);
478 }

514 cparser_result_t
515 cparser_cmd_save_roster_filename (cparser_context_t *context, char **filename)
516 {
517  FILE *fp;
518  int n;
519 
520  assert(context && filename);

537  if (roster[n].bonus_factor) {

539  }
540  if (roster[n].pc_ip_addr) {
541  fprintf(fp, " pc-ip-address %d.%d.%d.%d\n",
542  (int)((roster[n].pc_ip_addr >> 24) & 0xff),
543  (int)((roster[n].pc_ip_addr >> 16) & 0xff),
544  (int)((roster[n].pc_ip_addr >> 8) & 0xff),
545  (int)((roster[n].pc_ip_addr >> 0) & 0xff));
546  }
547  if (roster[n].dob.month && roster[n].dob.day && roster[n].dob.year) {
548  fprintf(fp, " date-of-birth %d %d %d\n", roster[n].dob.month,
549  roster[n].dob.day, roster[n].dob.year);
550  }
551  if (strlen(roster[n].title)) {
552  fprintf(fp, " title %s\n", roster[n].title);
553  }
554  }
555 
556  fclose(fp);
557  return CPARSER_OK;
558 }
