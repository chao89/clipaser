119 int
120 test_state_transition (const char *input, const cparser_state_t final_state,
121  const short token_tos, const cparser_token_t *tokens,
122  const short current_pos, const short last_good)
123 {
124  cparser_t parser;
125  int n;
126 
127  test_init_parser(&parser);
128 
129  if (CPARSER_STATE_WHITESPACE != parser.state) {
130  return 0;
131  }
132  if (!test_input(&parser, input)) {
133  return 0;
134  }
135  if (final_state != parser.state) {
136  printf("ERROR: Unexpected state. (expected:%d got:%d)\n",
137  final_state, parser.state);
138  return 0;
139  }
140  /* Check against expected token stack */
141  if (token_tos != parser.token_tos) {
142  printf("ERROR: Unexpected token stack size. (expected:%d got:%d)\n",
143  token_tos, parser.token_tos);
144  return 0;
145  }
146  for (n = 0; n <= token_tos; n++) {
147  if (tokens[n].token_len != parser.tokens[n].token_len) {
148  printf("ERROR: Token %d length mismatch.\n", n);
149  return 0;
150  }
151  if (tokens[n].begin_ptr != parser.tokens[n].begin_ptr) {
152  printf("ERROR: Token %d begin pointer mismatch.\n", n);
153  return 0;
154  }
155  if (tokens[n].parent != parser.tokens[n].parent) {
156  printf("ERROR: Token %d parent pointer mismatch.\n", n);
157  return 0;
158  }
159  if (tokens[n].token_len &&
160  strcmp(tokens[n].buf, parser.tokens[n].buf)) {
161  printf("ERROR: Token %d buffer mismatch.\n", n);
162  return 0;
163  }
164  }
165  if (current_pos != parser.current_pos) {
166  printf("ERROR: Current position is incorrect (expect:%d got:%d).\n",
167  current_pos, parser.current_pos);
168  return 0;
169  }
170  if (last_good != parser.last_good) {
171  printf("ERROR: Last good position is incorrect (expect:%d got:%d).\n",
172  last_good, parser.last_good);
173  return 0;
174  }
175  return 1;
176 }

353 int
354 main (int argc, char *argv[])
355 {
356  int n, rc, num_passed = 0, num_failed = 0;
357  struct {
358  char *name;
359  test_fn_t test_fn;
360  } testcases[] = {
361  /* These are the normal transition for accepting a valid command */
362  { "(WHITESPACE, SPACE) -> WHITESPACE", test_whitespace_space_whitespace },
363  { "(WHITESPACE, CHAR) -> TOKEN", test_whitespace_char_token },
364  { "(TOKEN, CHAR) -> TOKEN", test_token_char_token },
365  { "(TOKEN, SPACE) -> WHITESPACE", test_token_space_whitespace },
366 
367  /* These are transitions that lead to errors */
368  { "(WHITESPACE, CHAR) -> ERROR", test_whitespace_char_error },
369  { "(TOKEN, CHAR) -> ERROR", test_token_char_error },
370  { "(TOKEN, SPACE) -> ERROR", test_token_space_error },
371  { "(ERROR, SPACE) -> ERROR", test_error_space_error },
372  { "(ERROR, CHAR) -> ERROR", test_error_char_error },
373 
374  /* These are transitions involved in erase */
375  { "(WHITESPACE, ERASE) -> WHITESPACE", test_whitespace_erase_whitespace },
376  { "(WHITESPACE, ERASE) -> TOKEN", test_whitespace_erase_token },
377  { "(TOKEN, ERASE) -> TOKEN", test_token_erase_token },
378  { "(TOKEN, ERASE) -> WHITESPACE", test_token_erase_whitespace },
379  { "(ERROR, ERASE) -> ERROR", test_error_erase_error },
380  { "(ERROR, ERASE) -> WHITESPACE", test_error_erase_whitespace },
381  { "(ERROR, ERASE) -> TOKEN", test_error_erase_token }
382  };
383  for (n = 0; n < (sizeof(testcases)/sizeof(testcases[0])); n++) {
384  rc = testcases[n].test_fn();
385  printf("%s: %s\n", rc ? "PASS" : "FAIL", testcases[n].name);
386  if (rc) {
387  num_passed++;
388  } else {
389  num_failed++;
390  }
391  }
392 
393  printf("Total=%d Passed=%d Failed=%d\n", num_passed + num_failed,
394  num_passed, num_failed);
395  return 0;
396 }
