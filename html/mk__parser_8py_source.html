74  def __init__(self, node_type, param, desc, flags, list_kw=None):
75  '''
76  Constructor.
77  '''
78 
79  ## Node type.
80  self.type = node_type
81  ## Parameter name.
82  self.param = param
83  ## Node description
84  self.desc = ''
85  if desc:
86  self.desc = ' "%s",\n' % desc
87  else:
88  self.desc = ' NULL,\n'
89  ## Flags
90  self.flags = flags
91  ## List of children nodes
92  self.children = []
93  if list_kw == None:
94  self.list_kw = [] # used only for LIST tokens
95  else:
96  self.list_kw = list_kw
97 
98  # Cannot fill these out until we insert the node to the tree
99  ## Reference to parent node
100  self.parent = None
101  ## Depth of the node
102  self.depth = 0
103  self.path = ''
104  self.next = None
105  return

107  def add_child(self, child):
108  '''
109  Add a child node to a tree node.
110 
111  @param child A Node object that is added to this node as its child.
112 
113  @return The child node added.
114  '''
115  for c in self.children:
116  if (c.type == child.type) and (c.param == child.param):
117  # The node already exists. Re-use the existing node.
118  # But check if the hidden flag should be cleared. If the new
119  # node does not have CPARSER_NODE_HIDDEN.
120  if 'CPARSER_NODE_HIDDEN' not in child.flags:
121  try:
122  c.delete('CPARSER_NODE_HIDDEN')
123  except:
124  pass
125  return c
126  # Fill out some information that are tree structure dependent.
127  # These information are actually embedded in the tree already
128  # However, we compute them and cache them to reduce the
129  # processing time later. This approach increases the storage
130  # requirement but decreases the processing time.
131  child.parent = self
132  child.depth = self.depth + 1
133  if child.is_list():
134  # For LIST node, param is a dictionary.
135  child.path = self.path + '_' + child.param
136  elif child.is_param() or child.is_keyword():
137  child.path = self.path + '_' + child.param.replace('-','_')
138  elif 'END' == child.type:
139  child.path = self.path + '_eol'
140  elif 'ROOT' == child.type:
141  # If we are adding a child ROOT node, the parent ('self' here)
142  # must be an END node. For END node, we insert a string
143  # 'cparser_glue' to the front of self.path. So, we must
144  # remove it first before creating the submode root path
145  child.path = self.param.replace('cparser_glue', '') + '_root'
146  if len(self.children) > 0:
147  self.children[-1].next = child
148 
149  # Insert the node into the children list
150  self.children.append(child)
151  return child

225  def walk(self, fn, mode, cookie):
226  last = len(self.children)
227  count = 0
228  if (mode == 'pre-order') or (mode == 'func'):
229  if ((mode == 'pre-order') or
230  ((self.type == 'END') and
231  (not self.flags.count('CPARSER_NODE_FLAGS_OPT_PARTIAL')))):
232  fn(self, cookie)
233  count += 1
234  for n in range(last):
235  count += self.children[n].walk(fn, mode, cookie)
236  elif 'post-order' == mode:
237  for n in range(last):
238  count += self.children[last-1-n].walk(fn, mode, cookie)
239  fn(self, cookie)
240  count += 1
241  else:
242  raise ValueError, 'Unknown walk mode: %s' % mode
243  return count

245  def c_struct(self):
246  '''
247  Generate the C structure name.
248 
249  @return Return a string that contains the C structure for the node.
250  '''
251  msg = ''
252  if 'LIST' == self.type:
253  # For LIST token, build its nodes. First, build the variable name.
254  list_kw_len = len(self.list_kw)
255  for n in range(list_kw_len-1, -1, -1):
256  this_kw = self.list_kw[n]
257  if n == (list_kw_len - 1):
258  next = 'NULL' # last keyword in the list
259  else:
260  next = '&cparser_list_node%s_%s' % (self.path, self.list_kw[n+1].replace('-', '_'))
261  msg += 'cparser_list_node_t cparser_list_node%s_%s = {\n' % (self.path, this_kw.replace('-', '_'))
262  msg += ' %s,\n' % next
263  msg += ' "%s"\n' % this_kw
264  msg += ' };\n\n'
265 
266  if self.parent == None:
267  msg += 'cparser_node_t cparser_root = {\n'
268  else:
269  msg += 'cparser_node_t cparser_node%s = {\n' % self.path
270  # type
271  msg += ' CPARSER_NODE_%s,\n' % self.type
272  # flags
273  if len(self.flags) == 0:
274  msg += ' 0,\n'
275  else:
276  msg += ' ' + ' | '.join(self.flags) + ',\n'
277  # param
278  if 'ROOT' == self.type: msg += ' NULL,\n'
279  elif 'END' == self.type: msg += ' %s,\n' % self.param
280  elif 'KEYWORD' == self.type: msg += ' "%s",\n' % self.param
281  elif 'LIST' == self.type:
282  msg += ' &cparser_list_node%s_%s,\n' % (self.path, self.list_kw[0].replace('-', '_'))
283  else: msg += ' "<%s:%s>",\n' % (self.type, self.param)
284  # desc
285  msg += self.desc
286  # sibling
287  if self.next:
288  msg += ' &cparser_node%s,\n' % self.next.path
289  else:
290  msg += ' NULL,\n'
291  # children
292  if len(self.children) > 0:
293  msg += ' &cparser_node%s\n' % self.children[0].path
294  else:
295  msg += ' NULL\n'
296  msg += '};\n\n'
297 
298  return msg

300  def walk_up_to_root(self):
301  '''
302  Return a list of Node objects that forms a path from root to this node.
303 
304  @return A list of Node objects that forms a path from root (first elemnt) to this node (last element).
305  '''
306  assert self.type == 'END'
307  p = []
308  cur_node = self
309  while cur_node.parent:
310  p.insert(0, cur_node)
311  cur_node = cur_node.parent
312  if cur_node.type == 'ROOT':
313  break
314  return p

316  def action_fn(self):
317  '''
318  Generate the action function prototype.
319 
320  @return The C action function prototype for a command.
321  '''
322  # Build a list of parse nodes that forms the path from the root.
323  # to this end node
324  path = self.walk_up_to_root()
325 
326  # Declare the action function
327  msg = ('cparser_result_t %s(cparser_context_t *context' %
328  self.param.replace('cparser_glue', 'cparser_cmd'))
329 
330  # Declare the variable list
331  for n in path:
332  if not n.is_param(): continue
333  msg += ',\n %s*%s_ptr' % (Node.TYPES[n.type], n.param)
334  msg += ');\n'
335  return msg

337  def glue_fn(self):
338  '''
339  Generate the glue funtion.
340 
341  @return The C glue function for a command.
342  '''
343  # Build a list of parse nodes that forms the path from the root.
344  # to this end node
345  path = self.walk_up_to_root()
346 
347  # Build the glue function
348  msg = 'cparser_result_t\n'
349  msg += '%s (cparser_t *parser)\n' % self.param
350  msg += '{\n'
351 
352  # Declare the variable list
353  skip = ''
354  for n in path:
355  if not n.is_param(): continue
356  skip = '\n'
357  val_type = Node.TYPES[n.type]
358  msg += ' %s%s_val;\n' % (val_type, n.param)
359  msg += ' %s*%s_ptr = NULL;\n' % (val_type, n.param)
360  if skip: msg += ' cparser_result_t rc;\n'
361  msg += skip
362 
363  # Extract the parameters
364  k = -1
365  for n in path:
366  k = k + 1
367  if not n.is_param(): continue
368  msg += (' rc = cparser_get_%s(&parser->tokens[%d], &%s_val);\n' %
369  (n.type.lower(), k, n.param))
370  if n.is_optional():
371  msg += ' if (CPARSER_OK == rc) {\n'
372  msg += ' %s_ptr = &%s_val;\n' % (n.param, n.param)
373  msg += ' } else {\n'
374  msg += ' assert(%d > parser->token_tos);\n' % (k+1)
375  msg += ' }\n'
376  else:
377  msg += ' assert(CPARSER_OK == rc);\n'
378  msg += ' %s_ptr = &%s_val;\n' % (n.param, n.param)
379 
380  # Call the user-provided action function
381  msg += (' %s(&parser->context' %
382  self.param.replace('cparser_glue', 'cparser_cmd'))
383  for n in path:
384  if n.is_param(): msg += ',\n %s_ptr' % n.param
385  msg += ');\n'
386  msg += ' return CPARSER_OK;\n'
387  msg += '}\n\n'
388  return msg

390 class Token:
391  '''Token class. This class represents a token in a CLI command.'''
392  ## Beginning of a parameter token
393  BEGIN = '^<'
394  ## End of a parameter token
395  END = '>$'
396  ## Token type
397  TYPE = '([A-Z][A-Z0-9]*)'
398  ## Keyword
399  KW = '[a-zA-Z0-9_-]+'
400  LIST_KW = '([^:]+)'
401  ## Parameter name
402  PARAM = '([a-zA-Z][a-zA-Z0-9_]*)'
403  ## Description of a node
404  DESC= '(:(.+))*'
405 
406  @classmethod
407  def valid_keyword(cls, s):
408  '''Check whether a string is a valid keyword.
409 
410  @return True if the string is a valid keyword; False otherwise.
411  '''
412  return bool(re.search('^' + cls.KW + '$', s))

414  def __init__(self, s):
415  '''Constructor.
416 
417  @param s Token string.
418  '''
419 
420  # Create the fields of this object
421  ## Node type
422  self.type = ''
423  ## Parameter name
424  self.param = ''
425  ## Description of the node
426  self.desc = ''
427  ## If it is a LIST node, list of keywords.
428  self.list_kw = []
429 
430  # Check if this is a keyword
431  if Token.valid_keyword(s):
432  self.type = 'KEYWORD'
433  self.param = s
434  self.desc = None
435  self.list_kw = []
436  return None
437 
438  # This token must be a parameter of some kind. Parse the type.
439  m = re.search(Token.BEGIN + Token.TYPE + ':(.+)' + Token.END, s)
440  if not m:
441  raise ValueError, 'Invalid token "%s".' % s
442  if m.group(1) not in Node.TYPES:
443  raise ValueError, 'Unknown token type "%s".' % m.group(1)
444  self.type = m.group(1)
445 
446  # Check if it is a LIST type.
447  if self.type == 'LIST':
448  m = re.search(Token.BEGIN + Token.TYPE + ':' + Token.LIST_KW + ':' +
449  Token.PARAM + Token.DESC + Token.END, s)
450  if not m:
451  raise ValueError, 'Malformed LIST token "%s".' % s
452  (self.type, list_kw, self.param, dummy, self.desc) = m.groups()
453  # Validate all keywords in the list
454  self.list_kw = list_kw.split(',')
455  for kw in self.list_kw:
456  if not self.valid_keyword(kw):
457  raise ValueError, 'Invalid LIST keyword "%s".' % kw
458  return None
459 
460  # Handle the rest of the parameters
461  m = re.search(Token.BEGIN + Token.TYPE + ':' + Token.PARAM + Token.DESC + Token.END, s)
462  if not m:
463  m = re.search(Token.BEGIN + Token.TYPE + ':([^:>]+)' + Token.DESC + Token.END, s)
464  assert m
465  raise ValueError, 'Invalid parameter name "%s".' % m.group(2)
466  (self.type, self.param, dummy, self.desc) = m.groups()
467  self.list_kw = []

477 def add_cli(root, line, comment):
478  global end_node
479  nodes = []
480  flags = []
481  hidden_flag = []
482  num_opt_start = 0
483  num_opt_end = 0
484 
485  # Convert a line into a token list
486  line = line.replace('\n','')
487  DBG('\n LINE: ', line)
488  tokens = line.split(' ')
489 
490  # Delete all token that is ''
491  cnt = tokens.count('')
492  for k in range(0, cnt):
493  tokens.remove('')
494  if len(tokens) == 0:
495  return root # this is a blank line. quit
496 
497  # If the '+' marker is with the first token, separate them
498  if tokens[0] == '+':
499  tokens.pop(0)
500  hidden_flag = [ 'CPARSER_NODE_FLAGS_HIDDEN', ]
501  elif tokens[0][0] == '+':
502  tokens[0] = tokens[0][1:]
503  hidden_flag = [ 'CPARSER_NODE_FLAGS_HIDDEN', ]
504 
505  # Convert tokens to parse tree nodes. '{' and '}' do not produce tree
506  # nodes. But they do affect the flags used in some nodes.
507  start_flag = False
508  for t in tokens:
509  # Parse each token
510  # Look for '{'
511  if '{' == t:
512  # In the last node, its flags field is a reference to
513  # 'flags'. So, if we append to it, the last node will get
514  # a new flag.
515  start_flag = True
516  num_opt_start = num_opt_start + 1
517  continue
518  # Look for '}'
519  if '}' == t:
520  # See comment in '{' case.
521  assert(len(nodes) > 0)
522  nodes[-1].flags.append('CPARSER_NODE_FLAGS_OPT_END')
523  num_opt_end = num_opt_end + 1
524  if num_opt_end == num_opt_start:
525  nodes[-1].flags.remove('CPARSER_NODE_FLAGS_OPT_PARTIAL')
526  continue
527 
528  if num_opt_start > num_opt_end:
529  flags = hidden_flag + ['CPARSER_NODE_FLAGS_OPT_PARTIAL',]
530  else:
531  flags = hidden_flag[:]
532 
533  if start_flag:
534  flags.append('CPARSER_NODE_FLAGS_OPT_START')
535 
536  # Get the token type
537  tt = Token(t)
538  nodes.append(Node(tt.type, tt.param, tt.desc, flags[:], tt.list_kw))
539  start_flag = False
540 
541  # hack alert - Check that if there are optional parameters, the format is ok
542 
543  DBG('TOKENS: ', tokens)
544  if debug:
545  sys.stdout.write(' NODES: ')
546  for nn in nodes[:-1]:
547  nn.display()
548  sys.stdout.write('\n ')
549  nodes[-1].display()
550  sys.stdout.write('\n')
551 
552  # We need to create the glue function name. Since the path of the node
553  # is set up when the node is inserted, we don't have it here. So,
554  # we manually walk all the nodes and generate the glue function name
555  glue_fn = 'cparser_glue' + root.param
556  for n in nodes:
557  glue_fn = glue_fn + '_' + n.param.replace('-','_')
558 
559  # Insert them into the parse tree
560  for k in range(0, num_opt_start+1):
561  DBG(' CLI: ', k)
562  num_braces = 0
563  cur_node = root
564  if num_opt_start == k:
565  end_node = Node('END', glue_fn, comment, hidden_flag[:])
566  else:
567  end_node = Node('END', glue_fn, None, ['CPARSER_NODE_FLAGS_OPT_PARTIAL',] + hidden_flag)
568  for n in nodes:
569  if n.flags.count('CPARSER_NODE_FLAGS_OPT_START'):
570  if num_braces == k:
571  num_braces += 1
572  break
573  num_braces += 1
574  if debug:
575  sys.stdout.write(' ')
576  cur_node.display()
577  sys.stdout.write('-> ')
578  n.display()
579  sys.stdout.write('\n')
580  cur_node = cur_node.add_child(n)
581  if debug:
582  sys.stdout.write(' ')
583  cur_node.display()
584  sys.stdout.write('-> ')
585  end_node.display()
586  sys.stdout.write('\n')
587  cur_node.add_child(end_node)
588  return root

594 def process_cli_file(filename, root, mode, labels, last_cli_root=None, last_cli_end=None, submode=False):
595  '''
596  Process one .cli file. This includes handling all preprocessors directive.
597 
598  @param filename Name of the .cli file.
599  @param root Root Node object of the parse tree.
600  @param mode "compile", "preprocess" or "mkdep"
601  @param labels A dictionary containing all defined labels used in
602  preprocessing.
603  @param last_cli_root The most recent root node created.
604  @param last_cli_end The most recent end node visited.
605  @param submode A boolean of whether we are in a submode.
606 
607  @return Return the new root node.
608  '''
609 
610  num_disable = 0
611  label_stack = []
612  deplist = []
613  comment = None
614  line = ''
615  try:
616  fin = open(filename, 'r') except:
617  print 'Fail to open %s.' % filename
618  sys.exit(-1)
619 
620  line_num = 0
621  for line in fin:
622  line_num = line_num + 1
623 
624  # Process the file line by line. The orderof processing
625  # These different directives is extremely important. And the
626  # order below is not arbitrary.
627  #
628  # We must process #endif not matter what. Otherwise, once
629  # we start a #ifdef block that is omitted, we'll never be able
630  # to terminate it. Then, we omit every other type of line as long
631  # as there is at least one disable #ifdef left. Afterward, we
632  # check for illegal directives. A normal command line is handled
633  # last. Illegal tokens are checked inside add_cli().
634 
635  # #endif
636  m = re.search('^#endif', line)
637  if m:
638  if len(label_stack) == 0:
639  print('%s:%d: Unmatched #ifdef/#ifndef' % (filename, line_num))
640  sys.exit(-1)
641  num_disable = num_disable - label_stack.pop(0)[1]
642  continue
643  # Skip the rest of processing because some #ifdef/#ifndef is
644  # keeping this line from being processed.
645  if (num_disable > 0):
646  continue
647  # Check for illegal preprocessor directives
648  if (re.search('^#', line) and
649  (not re.search('^#ifdef(\S*\/\/.*)*', line) and
650  not re.search('^#submode(\S*\/\/.*)*', line) and
651  not re.search('^#endsubmode(\S*\/\/.*)*', line) and
652  not re.search('^#include(\S*\/\/.*)*', line))):
653  print('%s:%d: Unknown preprocessor directive.' % (filename, line_num))
654  sys.exit(-1)
655  # Comment
656  m = re.search('^\s*\/\/\s*(.*)', line)
657  if m:
658  if 'compile' == mode:
659  comment = m.group(1)
660  if 'preprocess' == mode:
661  sys.stdout.write(line)
662  continue
663  # #ifdef
664  m = re.search('^#ifdef (.+)', line)
665  if m:
666  l = m.group(1)
667  val = 0
668  if not labels.has_key(l):
669  val = 1
670  num_disable = num_disable + 1
