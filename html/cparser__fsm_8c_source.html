45 int
46 cparser_match (const cparser_t *parser, const char *token, const int token_len,
47  cparser_node_t *parent, cparser_node_t **match, int *is_complete)
48 {
49  int num_matches = 0, local_is_complete;
50  cparser_node_t *child;
51  cparser_result_t rc;
52 
53  assert(token && parent && match && is_complete);
54  *match = NULL;
55  *is_complete = 0;
56  for (child = parent->children; NULL != child; child = child->sibling) {
57  if (!NODE_USABLE(parser, child)) {
58  continue;
59  }
60  local_is_complete = 0;
61  rc = cparser_match_fn_tbl[child->type](token, token_len, child,
62  &local_is_complete);
63  if (CPARSER_OK == rc) {
64  num_matches++;
65  /*
66  * Return only the highest priority match unless the lower
67  * priority match is complete and the higher one is only
68  * partially matched.
69  */
70  if (!(*match)) {
71  /* Return only the highest priority match */
72  *match = child;
73  *is_complete = local_is_complete;
74  } else {
75  if (!(*is_complete) && local_is_complete) {
76  *match = child;
77  *is_complete = local_is_complete;
78  }
79  }
80  }
81  }
82 
83 #ifdef SHORTEST_UNIQUE_KEYWORD
84  /*
85  * There is a only one match and it is a keyword. Consider this a
86  * complete match.
87  */
88  if ((1 == num_matches) && (CPARSER_NODE_KEYWORD == (*match)->type)) {
89  *is_complete = 1;
90  }
91 #endif /* SHORTEST_UNIQUE_KEYWORD */
92  return num_matches;
93 }

103 cparser_token_stack_reset (cparser_t *parser)
104 {
105  int n;
106  cparser_token_t *token;
107 
108  assert(VALID_PARSER(parser));

137 static cparser_state_t
138 cparser_ws_erase (cparser_t *parser, const char ch, int *ch_processed)
139 {
140  cparser_token_t *token;
141 
142  assert(parser && ch_processed);
143  if (parser->current_pos > 0) {
144  *ch_processed = 1;
145  parser->current_pos--;
146  if (0 < parser->token_tos) {
147  token = &parser->tokens[parser->token_tos-1];
148  if (token->begin_ptr + token->token_len >= parser->current_pos) {
149  parser->cur_node = token->parent;
150  token->parent = NULL;
151 
152  /* Pop the token on top of the stack */
153  token = CUR_TOKEN(parser);
154  token->begin_ptr = -1;
155  token->token_len = 0;
156  token->parent = NULL;
157  token->node = NULL;
158  token->buf[0] = '\0';
159  parser->token_tos--;
160  return CPARSER_STATE_TOKEN;
161  }
162  }
163  } else {
164  *ch_processed = 0;
165  }
166  return CPARSER_STATE_WHITESPACE;
167 }

204 static cparser_state_t
205 cparser_ws_char (cparser_t *parser, const char ch, int *ch_processed)
206 {
207  cparser_node_t *match;
208  int is_complete;
209  cparser_token_t *token;
210 
211  assert(parser && ch_processed);
212  *ch_processed = 1;
213 
214  if (!cparser_match(parser, &ch, 1, parser->cur_node, &match, &is_complete)) {
215  return CPARSER_STATE_ERROR; /* no token match */
216  }
217 
218  token = CUR_TOKEN(parser);
219  token->begin_ptr = parser->current_pos;
220 
221  /* A valid token found. Add to token stack */
222  INSERT_TOK_STK(token, ch);
223  return CPARSER_STATE_TOKEN;

271 static cparser_state_t
272 cparser_tok_space (cparser_t *parser, const char ch, int *ch_processed)
273 {
274  cparser_node_t *match;
275  int is_complete;
276  cparser_token_t *token;
277 
278  assert(parser && (' ' == ch) && ch_processed);
279  *ch_processed = 1;
280  token = CUR_TOKEN(parser);
281  if ((1 <= cparser_match(parser, token->buf, token->token_len,
282  parser->cur_node, &match, &is_complete)) &&
283  (is_complete)) {
284  /* Save the parent node for this token and "close" the token */
285  token->parent = parser->cur_node;
286  token->node = match;
287  token->buf[token->token_len] = '\0';
288 
289  /* Push it into the stack */
290  parser->token_tos++;
291  assert(CPARSER_MAX_NUM_TOKENS > parser->token_tos);
292  token = CUR_TOKEN(parser);
293  assert(-1 == token->begin_ptr);
294  assert(0 == token->token_len);
295  assert(NULL == token->parent);
296 
297  parser->cur_node = match;
298 
299  return CPARSER_STATE_WHITESPACE;
300  }
301  return CPARSER_STATE_ERROR;
302 }

317 static cparser_state_t
318 cparser_tok_char (cparser_t *parser, const char ch, int *ch_processed)
319 {
320  cparser_node_t *match;
321  int is_complete;
322  cparser_token_t *token;
323 
324  assert(parser && ch_processed);
325  *ch_processed = 1;
326 
327  token = CUR_TOKEN(parser);
328  if (token->token_len < CPARSER_MAX_TOKEN_SIZE) {
329  INSERT_TOK_STK(token, ch);
330  } else {
331  return CPARSER_STATE_ERROR;
332  }
333  if (!cparser_match(parser, token->buf, token->token_len, parser->cur_node,
334  &match, &is_complete)) {
335  DELETE_TOK_STK(token);
336  return CPARSER_STATE_ERROR;
337  }
338 
339  return CPARSER_STATE_TOKEN;
340 }

352 static cparser_state_t
353 cparser_err_erase (cparser_t *parser, const char ch, int *ch_processed)
354 {
355  assert(parser && ch_processed);
356  *ch_processed = 1;
357  assert(0 < parser->current_pos); /* impossible to get to error state
358  * on an empty string.
359  */
360  parser->current_pos--;
361  if ((parser->last_good + 1) == parser->current_pos) {
362  cparser_token_t *token;
363  token = CUR_TOKEN(parser);
364  if (token->begin_ptr + token->token_len >= parser->current_pos) {
365  return CPARSER_STATE_TOKEN;
366  }
367  return CPARSER_STATE_WHITESPACE;
368  }
369  return CPARSER_STATE_ERROR;
370 }

420 cparser_result_t
421 cparser_fsm_input (cparser_t *parser, char ch)
422 {
423  int input_type, ch_processed, n, m;
424 
425  assert((uint32_t)parser->state < CPARSER_MAX_STATES);
426  /*
427  * We classify the input into one of 3 classes: backspace (BS),
428  * whitespace (SPC), regular characters (CHAR). We also check
429  * if the buffer is empty when erasing or the buffer is full
430  * when inserting. A more systematic approach is to check
431  * this in 9 state-input functions. But checking here reduces
432  * the amount of code by a good amount.
433  */
434  if ((parser->cfg.ch_erase == ch) || (parser->cfg.ch_del == ch)) {
435  /*
436  * Line buffer code already checks that there is character
437  * in token stack
438  */
439  assert(0 < parser->current_pos);
440  input_type = 0;
441  } else {
442  if (parser->current_pos >= (CPARSER_MAX_LINE_SIZE-1)) {
443  parser->cfg.printc(parser, '\a');
444  return CPARSER_OK;
445  }
446  if (' ' == ch) {
447  input_type = 1;
448  } else {
449  input_type = 2;
450  }
451  }
452  ch_processed = 0;
453  parser->state =
454  cparser_state_func_tbl[parser->state][input_type](parser, ch,
455  &ch_processed);
456 
457  if (ch_processed) {
458  if (0 != input_type) {
459  parser->current_pos++;
460  }
461  if (CPARSER_STATE_ERROR != parser->state) {
462  parser->last_good = parser->current_pos - 1;
463  }
464  }
465 
466  if (parser->cfg.flags & CPARSER_FLAGS_DEBUG) {
467  parser->cfg.printc(parser, '\n');
468 
469  /* Print out the state */
470  switch (parser->state) {
471  case CPARSER_STATE_WHITESPACE:
472  {
473  parser->cfg.prints(parser, "State: WHITESPACE\n");
474  break;
475  }
476  case CPARSER_STATE_TOKEN:
477  {
478  parser->cfg.prints(parser, "State: TOKEN\n");
479  break;
480  }
481  case CPARSER_STATE_ERROR:
482  {
483  parser->cfg.prints(parser, "State: ERROR\n");
484  break;
485  }
486  default:
487  parser->cfg.prints(parser, "State: UNKNOWN\n");
488  }
489 
490  /* Print out the parser internal buffer and token stack */
491  for (n = 0; n < parser->current_pos; n++) {
492  parser->cfg.printc(parser, cparser_line_char(parser, n));
493  }
494  parser->cfg.printc(parser, '\n');
495 
496  for (n = 0; n <= parser->token_tos; n++) {
497  parser->cfg.printc(parser, '[');
498  for (m = 0; m <= parser->tokens[n].token_len; m++) {
499  parser->cfg.printc(parser, parser->tokens[n].buf[m]);
500  }
501  parser->cfg.printc(parser, ']');
502  parser->cfg.printc(parser, '\n');
503  }
504 
505  /* Print the line buffer again */
506  cparser_line_print(parser, 1, 1);
507  }
508  return CPARSER_OK;
509 }
